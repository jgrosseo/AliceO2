33 void SVertexer::process(const gsl::span<const PVertex>& vertices, // primary vertices
34  const gsl::span<const GIndex>& trackIndex, // Global ID's for associated tracks
35  const gsl::span<const VRef>& vtxRefs, // references from vertex to these track IDs
36  const o2::globaltracking::RecoContainer& recoTracks, // accessor to various tracks
37  std::vector<V0>& v0s, // found V0s
38  std::vector<RRef>& vtx2V0refs // references from PVertex to V0
39 )
40 {
41  std::unordered_map<uint64_t, int> cache; // cache for tested combinations, the value >0 will give the entry of prevalidated V0 in the v0sTmp
42  std::vector<V0> v0sTmp(1); // 1st one is dummy!
43  std::vector<int> v0sIdx; // id's in v0sTmp used attached to p.vertices
44  std::vector<RRef> pv2v0sRefs; // p.vertex to v0 index references
45  std::vector<char> selQ(trackIndex.size(), 0);
46 
47  mPVertices = vertices;
48  mV0sTmp[0].clear();
49  auto tracksPool = buildT2V(trackIndex, vtxRefs, recoTracks);
50  // V0 finding
51  int ntr = tracksPool.size();
52 #ifdef WITH_OPENMP
53  omp_set_num_threads(mNThreads);
54  int dynGrp = std::min(4, std::max(1, mNThreads / 2));
55 #pragma omp parallel for schedule(dynamic, dynGrp)
56 #endif
57  for (int it0 = 0; it0 < ntr; it0++) {
58  auto& seed0 = tracksPool[it0];
59  for (int it1 = it0 + 1; it1 < ntr; it1++) {
60  auto& seed1 = tracksPool[it1];
61  if (seed1.vBracket > seed0.vBracket) { // all vertices compatible with seed1 are in future wrt that of seed0
62  break;
63  }
64  if (seed0.getSign() == seed1.getSign()) { // make sure the seeds have opPite charges
65  continue;
66  }
67  auto seedP = &seed0, seedN = &seed1;
68  if (seed0.getSign() < seed1.getSign()) { // order to + , -
69  std::swap(seedP, seedN);
70  }
71 #ifdef WITH_OPENMP
72  int ith = omp_get_thread_num();
73 #else
74  int ith = 0;
75 #endif
76  checkV0Pair(*seedP, *seedN, ith);
77  }
78  }
79 #ifdef WITH_OPENMP
80  for (int i = 1; i < mNThreads; i++) { // merge results of all threads
81  mV0sTmp[0].insert(mV0sTmp[0].end(), mV0sTmp[i].begin(), mV0sTmp[i].end());
82  mV0sTmp[i].clear();
83  }
84 #endif
85 
86  vtx2V0refs.clear();
87  vtx2V0refs.resize(vertices.size());
88  finalizeV0s(v0s, vtx2V0refs);
89 }

132 std::vector<SVertexer::TrackCand> SVertexer::buildT2V(const gsl::span<const GIndex>& trackIndex, // Global ID's for associated tracks
133  const gsl::span<const VRef>& vtxRefs, // references from vertex to these track IDs
134  const o2::globaltracking::RecoContainer& recoTracks) // accessor to various tracks
135 {
136  // build track->vertices from vertices->tracks, rejecting vertex contributors
137 
138  // track selector: at the moment reject prompt tracks contributing to vertex fit and unconstrained TPC tracks
139  auto selTrack = [&](GIndex gid) {
140  return (gid.isPVContributor() || !recoTracks.isTrackSourceLoaded(gid.getSource())) ? false : true;
141  };
142 
143  std::vector<TrackCand> tracksPool;
144  std::unordered_map<GIndex, int> tmap;
145  int nv = vtxRefs.size();
146  for (int iv = 0; iv < nv; iv++) {
147  const auto& vtref = vtxRefs[iv];
148  int it = vtref.getFirstEntry(), itLim = it + vtref.getEntries();
149  for (; it < itLim; it++) {
150  auto tvid = trackIndex[it];
151  if (!selTrack(tvid)) {
152  continue;
153  }
154  std::decay_t<decltype(tmap.find(tvid))> tref{};
155  if (tvid.isAmbiguous()) {
156  auto tref = tmap.find(tvid);
157  if (tref != tmap.end()) {
158  tracksPool[tref->second].vBracket.setMax(iv); // this track was already processed with other vertex, account the latter
159  continue;
160  }
161  }
162 
163  tracksPool.emplace_back(TrackCand{recoTracks.getTrack(tvid), tvid, {iv, iv}});
164  if (tvid.isAmbiguous()) { // track attached to >1 vertex, remember that it was already processed
165  tmap[tvid] = tracksPool.size() - 1;
166  }
167  }
168  }
169  LOG(INFO) << "Collected " << tracksPool.size() << " seeds";
170  return std::move(tracksPool);
171 }

174 void SVertexer::finalizeV0s(std::vector<V0>& v0s, std::vector<RRef>& vtx2V0refs)
175 {
176  auto& tmpV0s = mV0sTmp[0];
177  int nv0 = tmpV0s.size();
178  std::vector<int> v0sortid(nv0);
179  std::iota(v0sortid.begin(), v0sortid.end(), 0);
180  std::sort(v0sortid.begin(), v0sortid.end(), [&](int i, int j) { return tmpV0s[i].getVertexID() < tmpV0s[j].getVertexID(); });
181  int pvID = -1, nForPV = 0;
182  for (int iv = 0; iv < nv0; iv++) {
183  const auto& v0 = tmpV0s[iv];
184  if (pvID < v0.getVertexID()) {
185  if (pvID > -1) {
186  vtx2V0refs[pvID].setEntries(nForPV);
187  }
188  pvID = v0.getVertexID();
189  vtx2V0refs[pvID].setFirstEntry(v0s.size());
190  nForPV = 0;
191  }
192  v0s.push_back(v0);
193  nForPV++;
194  }
195  if (pvID != -1) { // finalize
196  vtx2V0refs[pvID].setEntries(nForPV);
197  // fill empty slots
198  int ent = v0s.size();
199  for (int ip = vtx2V0refs.size(); ip--;) {
200  if (vtx2V0refs[ip].getEntries()) {
201  ent = vtx2V0refs[ip].getFirstEntry();
202  } else {
203  vtx2V0refs[ip].setFirstEntry(ent);
204  }
205  }
206  }
207 }

210 bool SVertexer::checkV0Pair(TrackCand& seedP, TrackCand& seedN, int ithread)
211 {
212  auto& fitter = mFitter2Prong[ithread];
213  int nCand = fitter.process(seedP, seedN);
214  if (nCand == 0) { // discard this pair
215  return false;
216  }
217  const auto& v0XYZ = fitter.getPCACandidate();
218  // check closeness to the beam-line
219  auto r2 = (v0XYZ[0] - mMeanVertex.getX()) * (v0XYZ[0] - mMeanVertex.getX()) + (v0XYZ[1] - mMeanVertex.getY()) * (v0XYZ[1] - mMeanVertex.getY());
220  if (r2 < mMinR2ToMeanVertex) {
221  return false;
222  }
223  if (!fitter.isPropagateTracksToVertexDone() && !fitter.propagateTracksToVertex()) {
224  return false;
225  }
226  auto& trPProp = fitter.getTrack(0);
227  auto& trNProp = fitter.getTrack(1);
228  std::array<float, 3> pP, pN;
229  trPProp.getPxPyPzGlo(pP);
230  trNProp.getPxPyPzGlo(pN);
231  // estimate DCA of neutral V0 track to beamline: straight line with parametric equation
232  // x = X0 + pV0[0]*t, y = Y0 + pV0[1]*t reaches DCA to beamline (Xv, Yv) at
233  // t = -[ (x0-Xv)*pV0[0] + (y0-Yv)*pV0[1]) ] / ( pT(pV0)^2 )
234  // Similar equation for 3D distance involving pV0[2]
235  std::array<float, 3> pV0 = {pP[0] + pN[0], pP[1] + pN[1], pP[2] + pN[2]};
236  float dx = v0XYZ[0] - mMeanVertex.getX(), dy = v0XYZ[1] - mMeanVertex.getY();
237  float pt2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1], prodXY = dx * pV0[0] + dy * pV0[1], tDCAXY = -prodXY / pt2V0;
238  float dcaX = dx + pV0[0] * tDCAXY, dcaY = dy + pV0[1] * tDCAXY, dca2 = dcaX * dcaX + dcaY * dcaY;
239  if (dca2 > mMaxDCAXY2ToMeanVertex) {
240  return false;
241  }
242  float p2V0 = pt2V0 + pV0[2] * pV0[2], ptV0 = std::sqrt(pt2V0);
243  // apply mass selections
244  float p2Pos = pP[0] * pP[0] + pP[1] * pP[1] + pP[2] * pP[2], p2Neg = pN[0] * pN[0] + pN[1] * pN[1] + pN[2] * pN[2];
245  bool goodHyp = false;
246  for (int ipid = 0; ipid < SVertexerParams::NPIDV0; ipid++) {
247  if (mV0Hyps[ipid].check(p2Pos, p2Neg, p2V0, ptV0)) {
248  goodHyp = true;
249  break;
250  }
251  }
252  if (!goodHyp) {
253  return false;
254  }
255  auto vlist = seedP.vBracket.getOverlap(seedN.vBracket); // indices of vertices shared by both seeds
256  if (vlist.isInvalid()) {
257  LOG(WARNING) << "Incompatible tracks: V0 " << seedP.vBracket.asString() << " | V1 " << seedN.vBracket.asString();
258  return false;
259  }
260  bool added = false;
261  auto bestCosPA = mMinCosPointingAngle;
262  for (int iv = vlist.getMin(); iv <= vlist.getMax(); iv++) {
263  const auto& pv = mPVertices[iv];
264  // check cos of pointing angle
265  float dz = v0XYZ[2] - pv.getZ(), cosPointingAngle = (prodXY + dz * pV0[2]) / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
266  if (cosPointingAngle < bestCosPA) {
267  continue;
268  }
269  if (added) { // V0 was already added, just reassign to this vertex
270  auto& v0 = mV0sTmp[ithread].back();
271  v0.setCosPA(cosPointingAngle);
272  v0.setVertexID(iv);
273  bestCosPA = cosPointingAngle;
274  } else {
275  std::array<float, 3> v0XYZF = {float(v0XYZ[0]), float(v0XYZ[1]), float(v0XYZ[2])};
276  auto& v0 = mV0sTmp[ithread].emplace_back(v0XYZF, pV0, trPProp, trNProp, seedP.gid, seedN.gid);
277  v0.setCosPA(cosPointingAngle);
278  v0.setVertexID(iv);
279  v0.setDCA(fitter.getChi2AtPCACandidate());
280  bestCosPA = cosPointingAngle;
281  added = true;
282  }
283  }
284 
285  return true;
286 }
